\section{Data}

\begin{enumerate}

\item 200 ft min distance, for safety parameters. Based on testing. \\

\item Lift is approximately 1.2 to 1.4 with flaps all the way out\\

\item Glide ratio is approximately 30:1\\

\item Glides over 100 miles at cruise speed.\\

\end{enumerate}

\section{Design}

\begin{enumerate}

\item Upward V-tail

\begin{enumerate}

    \item For increased clearance \\
    
    \item Coordinated yaw and roll (rather than adverse)\\
    
    \item Historically better design than a t-tail for UAVs\\
    
    \item Handles cross wind much better\\
 \end{enumerate}
 
\item Approx. 2 years to design the airfoil\\ 

\item Additional 2 years for prototype and design\\

\item Wanted to design everything to achieve their specific goals\\

\end{enmerate}
